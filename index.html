Criando um servidor/cliente Socket

08/04/2019 às 11:50


SERVIDOR
import socket
                    
HOST = '0.0.0.0'                 #endereço do servidor
PORT = 50007              # porta do servidor
s = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
s.bind((HOST, PORT))
s.listen(1)
conn, addr = s.accept()
print ('Connected by', addr)
while 1:
data = conn.recv(1024)
if not data: break
conn.sendall(data)
conn.close()

CLIENTE
import socket
    
HOST = ''    #endereço do serviço
PORT = 50007              #porta do servidor
s = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
s.connect((HOST, PORT))
s.sendall('Hello, world')
data = s.recv(1024)
s.close()
print ('Received', repr(data))

import numpy as np
import matplotlib.pyplot as plt
import math as mt
                   
a=float(input('Entre com o primeiro valor do intervalo fechado:'))
b=float(input('Entre com o segundo valor do intervalo fechado:'))
                    
                    
print('\nCalculando a area no intervalo [',a,',',b,']')
                   
def funcao(x):
    return x**3
                    
def integra(a,b):
    n=1000000
    area=0
    vetor_x=np.zeros((n+1))
    vetor_y=np.zeros((n+1))
                        
    vetor_x[0]=a
    vetor_x[n]=b
    vetor_y[0]=a**2
    vetor_y[n]=b**2
    u=1   
    delta=(b-a)/(n*1.0)
                        
for u in range(len(vetor_x)):
    vetor_x[u]=a+(u*delta)
    vetor_y[u]=funcao(vetor_x[u])        
    u+=1
                            
for i in range(len(vetor_x)-1):
    area+=(vetor_x[i+1]-vetor_x[i])*np.absolute(vetor_y[i+1])
    plt.plot(vetor_x,vetor_y,"b-")

return (area)
                                        
areaf=integra(a,b)
print ('A area da função no intervalo [',a,',',b,'] é de aproximadamente de',areaf)
